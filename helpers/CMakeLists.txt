  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_SOURCE_DIR}/core 
  ${CMAKE_BINARY_DIR}/core)

  smb4kmounthelper.cpp)

  KF6::AuthCore
  KF6::CoreAddons

  KF6::KIOCore)
